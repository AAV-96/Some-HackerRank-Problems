import java.io.*;
import java.util.*;

class Abc
{
	public static void main(String args[])
	{
		Scanner sc = new Scanner(System.in);
		int t,k=0;
		String n;
		char c;
		t = sc.nextInt();
		for (int i=0; i<t; i++){
			k=0;
			n = sc.next();
			for (int j=0; j<n.length(); j++){
				c = n.charAt(j);
				if (c == '4'){
					k++;
				}
			}
			System.out.println(k);
		}		
	}
}
