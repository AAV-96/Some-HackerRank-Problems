import java.io.*;
import java.util.Scanner;

class Abc
{
	public static void main(String args[])
	{
		Scanner sc = new Scanner(System.in);
		String s;
		s = sc.next();
		int i,j,k=0;
		int l = s.length();
		while (i != l){
			while(j != l){
				char a = s.charAt(i);
				char b = s.charAt(j);
				if (a == b){
					k++; i++; j++;
				}
			}
		}
		System.out.println(k);
	}
}
